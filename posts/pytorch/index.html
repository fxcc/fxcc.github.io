class MyLoss(nn.Module):
def __init__(self):
    super(MyLoss, self).__init__()

def forward(self, pred, truth):
    truth = torch.mean(truth,1)
    truth = truth.view(-1,2048)
    pred  = pred.view(-1,2048)
    return  torch.mean(torch.mean((pred-truth)**2,1),0)
